<script src="https://cdn.jsdelivr.net/npm/three@0.154.0/build/three.min.js"></script>
<script src="https://cdn.jsdelivr.net/npm/three@0.154.0/examples/js/controls/PointerLockControls.js"></script>
<script src="https://cdn.jsdelivr.net/npm/three@0.154.0/examples/js/loaders/GLTFLoader.js"></script>

<script>
  let camera, scene, renderer, controls;
  let sunLight, nightLight, isNight = false;
  let chilloutMusic = new Audio('music/chillout.mp3');
  chilloutMusic.loop = true;

  init();
  animate();

  function init() {
    scene = new THREE.Scene();

    camera = new THREE.PerspectiveCamera(75, window.innerWidth/window.innerHeight, 0.1, 1000);
    camera.position.set(0, 2, 8);

    renderer = new THREE.WebGLRenderer({antialias:true});
    renderer.setSize(window.innerWidth, window.innerHeight);
    document.body.appendChild(renderer.domElement);

    // Luz dÃ­a (sol)
    sunLight = new THREE.DirectionalLight(0xffffff, 1);
    sunLight.position.set(20, 40, 20);
    scene.add(sunLight);

    // Luz noche (apagada al inicio)
    nightLight = new THREE.PointLight(0xffaa55, 1.2, 50);
    nightLight.position.set(0, 5, 0);
    nightLight.visible = false;
    scene.add(nightLight);

    // Cargar modelo Patio.glb
    const loader = new THREE.GLTFLoader();
    loader.load('patio.glb', function(gltf) {
      scene.add(gltf.scene);
      document.getElementById("loading").style.display = "none";
    });

    // Controles estilo videojuego
    controls = new THREE.PointerLockControls(camera, document.body);
    document.body.addEventListener('click', () => controls.lock());
    scene.add(controls.getObject());

    const keyStates = {};
    document.addEventListener('keydown', (event) => keyStates[event.code] = true);
    document.addEventListener('keyup', (event) => keyStates[event.code] = false);

    function movePlayer(delta) {
      const speed = 5 * delta;
      if (keyStates['KeyW']) controls.moveForward(speed);
      if (keyStates['KeyS']) controls.moveForward(-speed);
      if (keyStates['KeyA']) controls.moveRight(-speed);
      if (keyStates['KeyD']) controls.moveRight(speed);
    }

    // Alternar noche/dÃ­a con tecla â€œNâ€
    window.addEventListener('keydown', (event) => {
      if (event.key.toLowerCase() === 'n') {
        isNight = !isNight;
        sunLight.visible = !isNight;
        nightLight.visible = isNight;
        if(isNight) {
          chilloutMusic.play();
        } else {
          chilloutMusic.pause();
          chilloutMusic.currentTime = 0;
        }
        console.log("Modo ahora:", isNight ? "ðŸŒ™ Noche" : "â˜€ DÃ­a");
      }
    });

    // AnimaciÃ³n con movimiento
    let prevTime = performance.now();
    function animateMove() {
      const time = performance.now();
      const delta = (time - prevTime) / 1000;
      movePlayer(delta);
      prevTime = time;
      requestAnimationFrame(animateMove);
    }
    animateMove();

    window.addEventListener('resize', ()=> {
      camera.aspect = window.innerWidth / window.innerHeight;
      camera.updateProjectionMatrix();
      renderer.setSize(window.innerWidth, window.innerHeight);
    });
  }

  function animate() {
    requestAnimationFrame(animate);
    renderer.render(scene, camera);
  }
</script>
